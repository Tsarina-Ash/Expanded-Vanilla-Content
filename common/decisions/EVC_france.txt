revival_of_paris = {
	is_shown = {
		exists = c:FRA
		this = c:FRA
		NOT = {
			has_variable = french_intervention_in_vietnam
		}		
	}
	possible = {
		this = c:FRA
		has_technology_researched = nationalism
	}
	
	when_taken = {
		s:STATE_ILE_DE_FRANCE.region_state:FRA = {
			add_modifier = {
				name = parisian_revival
			}
		}
		set_variable = french_intervention_in_vietnam		
	}

	ai_chance = {
		base = 200
	}
}
