han_becomes_accepted_in_china = {
	is_shown = {
		exists = c:CHI
		this = c:CHI
        NOT = { 
            has_variable = qing_han_is_primary_culture_now 
        }		
	}
	possible = {
		this = c:CHI
		has_technology_researched = pan-nationalism
		has_law = law_type:law_multicultural
		not = {
			is_subject = yes
		}
	}
	when_taken = {
		add_primary_culture = cu:han
        set_variable = qing_han_is_primary_culture_now
	}

	ai_chance = {
		base = 40
	}	
}

great_leap_foward_decision = {
	is_shown = {
        NOT = { 
            has_variable = chi_has_done_great_leap_foward 
        }	
		has_technology_researched = socialism
		ig:ig_trade_unions = {
			is_marginal = no
		}
	}
	
	possible = {
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 15
			}
		}
		ig:ig_trade_unions = {
			is_marginal = no
		}		
	}

	when_taken = {
		add_modifier = great_leap_foward_modifier
		add_radicals = {
			value = 0.05
		}		
		every_scope_state = {
			limit = {
				is_incorporated = yes
				is_split_state = no
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 15
				}
			}
			add_modifier = {
				name = great_leap_foward_state_modifier
			} 
		}

		set_variable = chi_has_done_great_leap_foward		
	}

	ai_chance = {
		base = 1
	}
}
