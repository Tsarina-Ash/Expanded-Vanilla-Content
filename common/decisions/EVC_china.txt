han_becomes_accepted_in_china = {
	is_shown = {
		exists = c:CHI
		this = c:CHI
	}
	possible = {
		this = c:CHI
		has_technology_researched = pan-nationalism
		has_law = law_type:law_multicultural
		not = {
			is_subject = yes
		}
	}
	when_taken = {
		add_primary_culture = cu:han
	}

	ai_chance = {
		base = 40
	}	
}
